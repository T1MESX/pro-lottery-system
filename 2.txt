        function saveState() {
            if(isRunning) return;
            const state = { 
                min: lastMin, max: lastMax, batch: lastBatch, exclude: lastExclude, 
                noRepeat: document.getElementById('noRepeat').checked, drawn: globalDrawnHistory, 
                drawCount: drawCount, logs: globalAuditLogs, historyHtml: document.getElementById('historyList').innerHTML, 
                theme: document.documentElement.getAttribute('data-theme') || 'light', currentStars: currentStars, 
                lastPrizeNote: lastPrizeNote, isCustomPool, customPoolOrigin, customPoolExcluded, presets,
                volume: document.getElementById('volSlider').value 
            };
            
            let saved = false;
            let attempts = 0;
            while (!saved && attempts < 50) {
                try {
                    localStorage.setItem('pro_lottery_state', JSON.stringify(state));
                    saved = true;
                } catch (e) {
                    attempts++;
                    if (globalAuditLogs.length > 0) {
                        globalAuditLogs.shift();
                        const hl = document.getElementById('historyList');
                        const items = hl.querySelectorAll('.history-item');
                        if (items.length > 0) { items[items.length - 1].remove(); }
                        state.logs = globalAuditLogs; state.historyHtml = hl.innerHTML;
                        console.warn("Storage quota exceeded. FIFO triggered: Oldest log dropped.");
                    } else {
                        console.error("Storage completely full. Cannot save.");
                        break;
                    }
                }
            }
        }

        document.getElementById('importListInput').addEventListener('change', function(e) {
            const file = e.target.files[0];
            if (!file) return;
            const reader = new FileReader();
            
            reader.onload = function(evt) {
                try {
                    const data = evt.target.result;
                    let parsedList = [];
                    
                    if (file.name.endsWith('.xlsx') || file.name.endsWith('.xls')) {
                        const workbook = XLSX.read(data, {type: 'binary'});
                        const firstSheetName = workbook.SheetNames[0];
                        const worksheet = workbook.Sheets[firstSheetName];
                        const json = XLSX.utils.sheet_to_json(worksheet, {header: 1});
                        for(let i=0; i<json.length; i++) {
                            if(json[i] && json[i][0] !== undefined && json[i][0] !== null && String(json[i][0]).trim() !== '') {
                                parsedList.push(String(json[i][0]).trim());
                            }
                        }
                    } else {
                        parsedList = data.split(/[\r\n,，\s]+/).map(s => s.trim()).filter(s => s !== '');
                    }

                    if (parsedList.length === 0) {
                        alert(currentLang === 'zh' ? "未发现有效数据，请检查文件内容！" : "No valid data found, please check the file!");
                        return;
                    }

                    let seen = new Set();
                    let dupes = new Set();
                    parsedList.forEach(item => {
                        if (seen.has(item)) dupes.add(item);
                        else seen.add(item);
                    });
                    
                    customPoolOrigin = Array.from(seen);
                    isCustomPool = true;
                    
                    if (dupes.size > 0) {
                        let dupeArr = Array.from(dupes);
                        let msg = currentLang === 'zh' ? 
                            `⚠️ 提示：发现并已自动去重 ${dupeArr.length} 种重复项。\n\n[重复项]:\n${dupeArr.slice(0,5).join(', ')}${dupeArr.length > 5 ? ' ...' : ''}` :
                            `⚠️ Note: Automatically removed ${dupeArr.length} duplicate types.\n\n[Duplicates]:\n${dupeArr.slice(0,5).join(', ')}${dupeArr.length > 5 ? ' ...' : ''}`;
                        alert(msg);
                    }

                    toggleListModeUI(true);
                    globalDrawnHistory = [];
                    drawCount = 0;
                    document.getElementById('numberDisplay').innerHTML = "?";
                    updateStatsBoard();

                    addSystemRecord(t('logListImport', {n: customPoolOrigin.length}));
                    saveState();

                } catch (error) {
                    console.error(error);
                    alert(currentLang === 'zh' ? "解析失败，请检查文件格式。建议使用单列无空行的 Excel 或 TXT 文件。" : "Parsing failed. Please ensure a valid single-column Excel or TXT.");
                }
            };

            if (file.name.endsWith('.xlsx') || file.name.endsWith('.xls')) {
                reader.readAsBinaryString(file);
            } else {
                reader.readAsText(file);
            }
            e.target.value = ''; 
        });

        async function generateThumbnail(sourceB64, maxW, maxH, q) {
            if (!sourceB64) return null;
            return new Promise((resolve) => {
                const img = new Image();
                img.onload = () => {
                    const canvas = document.createElement('canvas');
                    const ctx = canvas.getContext('2d');
                    let w = img.width; let h = img.height;
                    if (w > h) { if (w > maxW) { h *= maxW / w; w = maxW; } }
                    else { if (h > maxH) { w *= maxH / h; h = maxH; } }
                    canvas.width = w; canvas.height = h;
                    ctx.drawImage(img, 0, 0, w, h);
                    resolve(canvas.toDataURL('image/webp', q)); 
                };
                img.onerror = () => resolve(null);
                img.src = sourceB64;
            });
        }

        async function checkAndSaveActivePreset() {
            if (!activePresetId || isRunning) return;
            const p = presets.find(x => x.id === activePresetId);
            if (!p || p.drawn) return;

            const currentNote = document.getElementById('prizeNote').value.trim();
            const currentBatch = document.getElementById('batchNum').value;

            let isModified = false;
            if (currentNote !== p.note) { p.note = currentNote; isModified = true; }
            if (String(currentBatch) !== String(p.batch)) { p.batch = currentBatch; isModified = true; }
            if (currentStars !== p.stars) { p.stars = currentStars; isModified = true; }
            
            if (currentImageBase64 !== p.image) {
                let comp = await forceLowResCompress(currentImageBase64);
                p.image = comp;
                currentImageBase64 = comp; 
                isModified = true;
            }

            if (isModified) {
                renderPresets(); 
                saveState();     
            }
        }

        function rollbackToPrePresetState() {
            if (prePresetState) {
                document.getElementById('prizeNote').value = prePresetState.note;
                lastPrizeNote = prePresetState.note;
                document.getElementById('batchNum').value = prePresetState.batch;
                lastBatch = prePresetState.batch;
                currentStars = prePresetState.stars;
                renderStars(currentStars);
                currentImageBase64 = prePresetState.image;
                
                const box = document.getElementById('imgPreviewBox'); 
                const imgPreview = document.getElementById('previewImg');
                if (currentImageBase64) {
                    imgPreview.src = currentImageBase64;
                    box.style.display = 'block';
                } else {
                    currentImageBase64 = null;
                    imgPreview.src = '';
                    box.style.display = 'none';
                    document.getElementById('prizeImage').value = '';
                }
            }
        }

        async function handlePresetClick(id) {
            if (isRunning) return;

            if (activePresetId) {
                await checkAndSaveActivePreset();
            }

            if (activePresetId === id) {
                activePresetId = null;
                rollbackToPrePresetState();
                renderPresets();
                return;
            }

            if (!activePresetId) {
                prePresetState = {
                    note: document.getElementById('prizeNote').value.trim(),
                    batch: document.getElementById('batchNum').value,
                    stars: currentStars,
                    image: currentImageBase64
                };
            }

            loadPreset(id);
        }

        window.movePreset = function(event, id, dir) {
            event.stopPropagation();
            if(isRunning) return;
            const idx = presets.findIndex(x => x.id === id);
            if(idx < 0) return;
            if(dir === -1 && idx > 0) {
                [presets[idx], presets[idx-1]] = [presets[idx-1], presets[idx]];
            } else if(dir === 1 && idx < presets.length - 1) {
                [presets[idx], presets[idx+1]] = [presets[idx+1], presets[idx]];
            }
            renderPresets();
            saveState();
        };

        window.deletePreset = function(event, id) {
            event.stopPropagation();
            if(isRunning) return;
            presets = presets.filter(x => x.id !== id);
            if(activePresetId === id) {
                activePresetId = null;
                rollbackToPrePresetState();
            }
            addSettingRecord(t('logPresetDel'));
            renderPresets();
            saveState();
        };

        async function addPreset() {
            if(isRunning) return;
            if(presets.length >= 20) { alert(t('alertPresetMax')); return; }
            
            const note = document.getElementById('prizeNote').value.trim();
            const batch = document.getElementById('batchNum').value;
            const pName = note || (currentLang === 'zh' ? '未命名奖项' : 'Unnamed Prize');
            
            let compressedB64 = await forceLowResCompress(currentImageBase64);

            const presetObj = {
                id: 'p_' + Date.now(),
                note: note,
                stars: currentStars,
                batch: batch,
                image: compressedB64,
                drawn: false
            };

            presets.push(presetObj);
            addSettingRecord(t('logPresetAdd', {name: pName}));
            
            activePresetId = null;
            document.getElementById('prizeNote').value = '';
            lastPrizeNote = '';
            document.getElementById('batchNum').value = 1;
            lastBatch = 1;
            currentStars = 0;
            renderStars(0);
            currentImageBase64 = null;
            document.getElementById('previewImg').src = '';
            document.getElementById('imgPreviewBox').style.display = 'none';
            document.getElementById('prizeImage').value = '';
            prePresetState = { note: '', batch: 1, stars: 0, image: null };

            renderPresets();
            saveState();
        }

        function renderPresets() {
            const list = document.getElementById('presetList');
            
            document.getElementById('presetCounterZh').innerText = `(${presets.length}/20)`;
            document.getElementById('presetCounterEn').innerText = `(${presets.length}/20)`;

            if (presets.length === 0) {
                list.innerHTML = `<div style="text-align:center; color:var(--text-light); padding:20px; font-size:13px;" id="presetEmptyHint"><span class="lang-zh">暂无预设。在左侧填好奖品后点击上方添加。</span><span class="lang-en">No presets. Fill prize info and click add.</span></div>`;
                return;
            }
            
            list.innerHTML = '';
            presets.forEach((p, index) => {
                const li = document.createElement('li');
                li.className = `preset-item ${activePresetId === p.id ? 'active' : ''} ${p.drawn ? 'drawn' : ''}`;
                li.onclick = () => handlePresetClick(p.id);

                let starStr = p.stars > 0 ? `<span style="color:#f59e0b">${'★'.repeat(p.stars)}</span>` : '';
                let batchStr = currentLang === 'zh' ? `${p.batch} 人` : `${p.batch} Ppl`;
                let imgTag = p.image ? `<img src="${p.image}" class="preset-thumb">` : '';
                let displayName = p.note || (currentLang === 'zh' ? '未命名奖项' : 'Unnamed Prize');

                li.innerHTML = `
                    <button class="preset-badge-del" onclick="deletePreset(event, '${p.id}')" title="删除">×</button>
                    <div class="preset-seq">${index + 1}</div>
                    <div class="preset-info">
                        <div class="preset-name">${displayName}</div>
                        <div class="preset-sub">${starStr} <span class="preset-badge">${batchStr}</span> ${p.drawn ? '✅' : ''}</div>
                    </div>
                    ${imgTag}
                    <div class="preset-controls">
                        <div class="preset-move-group">
                            ${index > 0 ? `<button class="preset-move-btn" onclick="movePreset(event, '${p.id}', -1)" title="${currentLang==='zh'?'上移':'Move Up'}"><svg viewBox="0 0 24 24"><polyline points="18 15 12 9 6 15"></polyline></svg></button>` : `<div style="width:18px;height:18px;"></div>`}
                            ${index < presets.length - 1 ? `<button class="preset-move-btn" onclick="movePreset(event, '${p.id}', 1)" title="${currentLang==='zh'?'下移':'Move Down'}"><svg viewBox="0 0 24 24"><polyline points="6 9 12 15 18 9"></polyline></svg></button>` : `<div style="width:18px;height:18px;"></div>`}
                        </div>
                    </div>
                `;
                list.appendChild(li);
            });
        }

        function loadPreset(id) {
            const p = presets.find(x => x.id === id);
            if(!p || p.drawn) return;

            document.getElementById('prizeNote').value = p.note;
            lastPrizeNote = p.note;
            document.getElementById('batchNum').value = p.batch;
            lastBatch = p.batch;
            currentStars = p.stars;
            renderStars(currentStars);

            currentImageBase64 = p.image;
            const box = document.getElementById('imgPreviewBox'); 
            const imgPreview = document.getElementById('previewImg');
            if(currentImageBase64) {
                imgPreview.src = currentImageBase64;
                box.style.display = 'block';
            } else {
                currentImageBase64 = null; 
                imgPreview.src = ''; 
                box.style.display = 'none'; 
                document.getElementById('prizeImage').value = ''; 
            }

            activePresetId = id;
            renderPresets();
            
            const pName = p.note || (currentLang === 'zh' ? '未命名奖项' : 'Unnamed Prize');
            addYellowRecord(t('logPrize', {val: pName}));
        }

        function validatePositiveInt(val) { return /^\d+$/.test(val) && Number(val) > 0; }
        
        function setupNumberInput(id, getPrevValue, setPrevValue, isRange) {
            const el = document.getElementById(id);
            el.addEventListener('blur', function() {
                if (this.validity && this.validity.badInput) {
                    alert(t('alertPositive')); 
                    this.value = getPrevValue(); 
                    return;
                }
                let val = this.value.trim();
                if (!validatePositiveInt(val) || val === "") { alert(t('alertPositive')); this.value = getPrevValue(); return; }
                
                if (isRange) {
                    let testMin = id === 'minNum' ? Number(val) : Number(document.getElementById('minNum').value);
                    let testMax = id === 'maxNum' ? Number(val) : Number(document.getElementById('maxNum').value);
                    if (testMax - testMin > 100000) {
                        alert(currentLang === 'zh' ? '⚠️ 性能保护：号码区间不能超过 10 万！' : '⚠️ Memory limit: Range cannot exceed 100,000!');
                        this.value = getPrevValue(); return;
                    }
                } 
                else if (id === 'batchNum' && Number(val) > 100) {
                    alert(currentLang === 'zh' ? '⚠️ 性能保护：单次抽取最多支持 100 人！' : '⚠️ Performance limit: Max batch size is 100!');
                    this.value = getPrevValue(); return;
                }

                if (val === String(getPrevValue())) return; 
                
                if (isRange && globalDrawnHistory.length > 0) {
                    if (confirm(t('alertReset'))) { 
                        performSilentReset(); 
                        setPrevValue(val);
                        updateStatsBoard(); 
                        const stats = getPoolStats(); 
                        addSystemRecord(t('logReset', {rem: stats.remaining, tot: stats.total}));
                        addSettingRecord(t('logRange', {min: lastMin, max: lastMax, rem: stats.remaining, tot: stats.total})); 
                        saveState();
                        return;
                    } 
                    else { this.value = getPrevValue(); return; }
                }
                
                setPrevValue(val);
                updateStatsBoard();
                if (isRange) { 
                    const stats = getPoolStats(); 
                    addSettingRecord(t('logRange', {min: lastMin, max: lastMax, rem: stats.remaining, tot: stats.total})); 
                }
                saveState();
                
                if (id === 'batchNum') checkAndSaveActivePreset();
            });
            el.addEventListener('keydown', function(e) { if (e.key === 'Enter') { e.preventDefault(); this.blur(); } });
        }
        setupNumberInput('minNum', () => lastMin, (v) => lastMin = v, true);
        setupNumberInput('maxNum', () => lastMax, (v) => lastMax = v, true);
